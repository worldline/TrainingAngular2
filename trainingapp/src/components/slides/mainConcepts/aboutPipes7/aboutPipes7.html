Main Concepts Pipes - Exisiting Ones - In your code

  • As services, Pipes are injectable in your components and services.
  • For now, you still need to add providers to use them
  • Then,call the transform method
import {Component} from '@angular/core'; import {LowerCasePipe} from '@angular/common'; import {BrowserModule} from '@angular/platform-browser'; @Component({ selector: 'app', template: ` <p> {{getLowerCase('aBcDeF')}}</p> `, providers: [LowerCasePipe] }) export class App { constructor(private lowercase: LowerCasePipe){} getLowerCase(word: string): string { return this.lowercase.transform(word); } } export const ModuleDefs= { bootstrap: [App], imports: [BrowserModule], declarations: [App] }
