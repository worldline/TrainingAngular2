Advanced Concepts Unit Tests - Testing services

  • Services are simple classes
  • They eventually benefit of Dependency Injection (through the Injector API)
  • We can then have to provide mocked service, such as Http for instance
  • Pay also attention to the done callback for asynchronous testing
import {Injectable} from '@angular/core'; import {Http, Response} from '@angular2/http'; import {Observable} from 'rxjs/Observable'; import 'rxjs/add/operator/map'; import {Book, Comment} from '../beans/book'; export interface Data{ content: string; } @Injectable() export class DataService{ constructor(private http: Http){} getData= (): Observable<Data> => { return this.http.get('/data.json') .map((response: Response) => { let data: Data = <Data> response.json(); return data; }); } } import {Injector} from '@angular/core'; import {Observable} from 'rxjs/Observable'; import 'rxjs/add/observable/of'; import {Http, Response, ResponseOptions} from 'angular2/http'; import {Data, DataService} from './dataService'; export function main(){ describe('DataService', () => { let parentInjector: Injector; let dataService: DataService; let data: Data; let response: Response; beforeEach(() => { data = { content:'some content' }; response = new Response(new ResponseOptions({ body: JSON.stringify(data) })); parentInjector = Injector.resolveAndCreate([ DataService, { provide: Http, useValue: { get: function() { return Observable.of(response); } } } ]); dataService = parentInjector.get(DataService); }); it('getData', (done:any) => { dataService.getData() .subscribe((currentData: Data) => { expect(currentData).toEqual(data); done(); }); }); }); }
